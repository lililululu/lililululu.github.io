Bing Background Change for Mac

  • By Biyou
  • 四 08 九月 2016

bing api

  • http://www.bing.com/HPImageArchive.aspx?format=js&idx=1&n=1
  • 只能爬当前日期的15个背景图片/国外和国内的图片有靴微的不一样
  • 搓搓的懒得套多线程,没有任何的try，仅供参考
  • save成jpg使用“wb”
#!/usr/bin/env python
# -*- coding: utf-8 -*-

import sys,os
reload(sys) 
sys.setdefaultencoding( "utf-8" )

import requests,re

def saveImage( imgUrl,imgName ="default.jpg" ):
    response = requests.get(imgUrl, stream=True)
    image = response.content
    DstDir=""
    try:
        with open(DstDir+imgName ,"wb") as jpg:
            jpg.write(image)     
            return
    except IOError:
        print("IO Error\n")
        return
    finally:
        print "[*] Download :" + imgName 
        jpg.close  

def geturl(url):

    r = requests.get(url)
    content = r.content
    if len(content) < 10 :
        return 
    p_url = re.compile(r'"url":"(.*?)"')
    p_date = re.compile(r'"startdate":"(.*?)"')
    bgurl = p_url.findall(content)[0]
    date = p_date.findall(content)[0]
    if not bgurl.startswith('http'):
        bgurl = 'http://www.bing.com' + bgurl
    return {"startdata":date ,"bgurl":bgurl}

if __name__ == '__main__':
    urls = []
    for i in range(20):
        url = "http://www.bing.com/HPImageArchive.aspx?format=js&idx=%s&n=1" %str(i)
        urls.append(url)
        s = {}
        s = geturl(url)
        if not s:
            continue
        saveImage(s['bgurl'],s['startdata']+'.jpg')

  • crontab 格式：分 时 日 月 星期 要运行的命令
crontab -e 
00 11 * * * python /Users/change_bg.py >> /Users/change_bg.log 2>&1 # 11:00
* */1 * * * /usr/local/apache/bin/apachectl restart #每小时
  • 使用osascript 来改变mac背景
  • tell application "Finder" set desktop picture to POSIX file "%s"
  • 必须是绝对路径，不然会报错错
#!/usr/bin/env python
# -*- coding: utf-8 -*-

#osascript -e 'tell application "Finder" to set desktop picture to POSIX file "/Library/Desktop Pictures/Earth Horizon.jpg"'
#http://www.bing.com/HPImageArchive.aspx?format=js&idx=0&n=1
"""
Change bg everyday~
"""

import sys,os
reload(sys) 
sys.setdefaultencoding( "utf-8" )

import requests,re
import subprocess

SCRIPT = """/usr/bin/osascript<<END
tell application "Finder"
set desktop picture to POSIX file "%s"
end tell
END"""

def saveImage( imgUrl,imgName ="today.jpg" ):
    response = requests.get(imgUrl, stream=True)
    image = response.content
    DstDir=""
    try:
        with open(DstDir+imgName ,"wb") as jpg:
            jpg.write( image)    

    except IOError:
        print("IO Error\n")
        return
    finally:
        print "[*] Download :" + imgName 
        jpg.close  
    #save also as today.jpg
    try:
        with open("today.jpg" ,"wb") as jpg:
            jpg.write( image)    
            return
    except IOError:
        print("IO Error\n")
        return
    finally:
        print "[*] Download :" + "today.jpg"
        jpg.close  

def geturl(url):
    r = requests.get(url)
    content = r.content
    if len(content) < 10 :
        return 
    p_url = re.compile(r'"url":"(.*?)"')
    p_date = re.compile(r'"startdate":"(.*?)"')
    bgurl = p_url.findall(content)[0]
    date = p_date.findall(content)[0]
    if not bgurl.startswith('http'):
        bgurl = 'http://www.bing.com' + bgurl
    return {"startdata":date ,"bgurl":bgurl}


def set_desktop_background(filename):
    subprocess.Popen(SCRIPT%filename, shell=True)

if __name__ == '__main__':
    urls = []
    url = "http://www.bing.com/HPImageArchive.aspx?format=js&idx=0&n=1"
    s = {}
    s = geturl(url)
    if not s :
        print "Connect Error"
    else:
        saveImage(s['bgurl'],s['startdata']+'.jpg')
        set_desktop_background('/Users/today.jpg')
