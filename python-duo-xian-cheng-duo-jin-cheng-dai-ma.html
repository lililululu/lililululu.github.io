Python 多线程／多进程 代码

  • By Biyou
  • 一 05 九月 2016
  • 进程优点：cpu密集型
  • 进程缺点：共享内存比较麻烦
  • 线程优点：共享内存方便/io密集型，网络访问可开多线程
  • 线程缺点：cpu密集型效果不佳

多线程

  • 使用threading 模块
  • t.setDaemon(True) 设置成后台程序，和thread.join()冲突，不可共用。join()方法使得线程可以等待另一个线程的运行，而setDaemon()方法使得线程在结束时不等待子线程。
# threading 1
import threading
import time

def func(tid):
  print "[*] Thread " + str(tid) 
  time.sleep(tid)
  print "[*] End Thread " + str(tid)

record = []
for i in range(20):
    thread = threading.Thread(target=func,args=(i,))
    #thread.setDaemon(True)
    thread.start()
    record.append(thread)

for thread in record:
    thread.join()

print "Done"

多进程

  • 使用multiprocessing 模块
#multiprocessing 1

import multiprocessing
import time

def func(tid):
  print "[*] Thread " + str(tid) 
  time.sleep(tid)
  print "[*] End Thread " + str(tid)

record = []
for i in range(20):
    process = multiprocessing.Process(target=func,args=(i,))
    process.start()
    record.append(process)

for thread in record:
    thread.join()
  • 使用Pool进程池。
  • 在进程join之前必须close进城池。
  • apply_async为非阻塞进程
  • res.get()用来获取返回值
#multiprocessing 2

import multiprocessing
import time,os

def func(msg,j):
  print "[*] Proceed " + str(j) + 'pid: ' + str(os.getpid())
  for i in xrange(3):

    time.sleep(1)
  print "[*] End proceed " + str(j) + 'pid: ' + str(os.getpid())
  return "done " + msg

if __name__ == "__main__":
  pool = multiprocessing.Pool(processes=10)
  result = []
  for i in xrange(20):
    msg = "hello %d" %(i)
    result.append(pool.apply_async(func, (msg,i )))
  pool.close()
  pool.join()
  for res in result:
    print res.get()
  print "Sub-process(es) done."

  • 多进程的共享，使用Value/Array对象进行
#! /usr/bin/env python 
# -*- coding: utf-8 -*-

import os
from multiprocessing import Process,Value,Array
"""
使用多进程共享内存对象进行通信，传输的对象是共享的
但在windows下不共享全局对象，这里最后那个测试代表了
通常的用例:分配工作
"""
procs=3
count=0

def showdata(label,val,arr):
        """在这个进程中打印数值"""
        msg='%-12s: pid:%d, global:%s, value:%s, array:%s'
        print(msg %(label,os.getpid(),count,val.value,list(arr)))

def updater(val,arr):
        """通过共享内存进行通信"""
        global count
        count +=1                               #全局计数器，非共享
        val.value +=1                           #传输的对象是共享的
        for i in range(3): arr[i] += 1

if __name__ == '__main__':
        scalar=Value('i',0)
        vector=Array('d',procs)                 #ctypes中的类型代码 就像int和double

        #在父进程中显示起始值
        showdata('parent start',scalar,vector)

        #派生子进程，传入共享内存
        p=Process(target=showdata,args=('child',scalar,vector))
        p.start()
        p.join()

        #传入父进程中更新过的共享内存，等待每次传入结束
        #每个子进程看到了父进程中到现在为止的args的更新(但全局变量的看不到)
        print('\nloop1 (updates in parent,serial children)...')
        for i in range(procs):
                count +=1
                scalar.value +=1
                vector[i] +=1
                p=Process(target=showdata,args=(('process %s' % i),scalar,vector))
                p.start()
                p.join()

        #同上，不允许子进程并行运行
        #所有进程都看到了最新一次迭代的结果，因为它们都共享这个对象
        print('\nloop2 (updates in parent,serial children)...')
        ps=[]
        for i in range(procs):
                count += 1
                scalar.value += 1
                vector[i] += 1
                p=Process(target=showdata,args=(('process %s' % i),scalar,vector))
                p.start()
                ps.append(p)
        for p in ps: p.join()

        #共享内存再派生子进程中进行更新，等待每个更新结束
        print('\nloop3 (updates in parent,serial children)...')
        for i in range(procs):
                p=Process(target=updater,args=(scalar,vector))
                p.start()
                p.join()
        showdata('parent temp',scalar,vector)

        #同上，不过允许子进程并行的更新

        ps=[]
        print('\nloop4 (updates in parent,serial children)...')
        for i in range(procs):
                p=Process(target=updater,args=(scalar,vector))
                p.start()
                ps.append(p)
        for p in ps: p.join()
        #尽在父进程中全局变量count=6

        #在此显示最终结果

        showdata('parent end',scalar,vector)
